void sort(int b[],int length)
{
for (int i = 0; i < length; i++)
{
for (int j = 0; j < length - i - 1; j++)
{
if (b[j] > b[j+1])
{
int t = b[j];
b[j] = b[j+1];
b[j+1] = t;
}
}
}
}
